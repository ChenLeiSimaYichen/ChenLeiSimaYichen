#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    string s1;
    cin>>s1;
    string s="";
    for(int i=0;i<s1.size()-1;i++){
        s+=s1[i];
    }
    string a=s;
    reverse(s.begin(),s.end());
    if(a==s){
        cout<<"TRUE";
    }
    else{
        cout<<"FALSE";
    }
    return 0;
}
